        function setupScrollNavigation() {
            // Écouter les clics sur les boutons de section pour faire défiler
            document.querySelectorAll('.section-button').forEach(button => {
                button.addEventListener('click', function() {
                    const sectionKey = this.getAttribute('data-section');
                    const sectionElement = document.getElementById(`section-${sectionKey}`);

                    if (sectionElement) {
                        // Calculer la position avec un offset pour éviter la superposition de l'en-tête et de la navigation
                        // sectionElement.offsetTop est la distance du haut de l'élément au haut du document
                        // On soustrait SCROLL_OFFSET_CLICK pour que le haut de la section se place SOUS les éléments fixes
                        const targetScrollPosition = sectionElement.offsetTop - SCROLL_OFFSET_CLICK;
                        window.scrollTo({
                            top: targetScrollPosition,
                            behavior: 'smooth'
                        });
                    }

                    // Mettre à jour l'état actif
                    document.querySelectorAll('.section-button').forEach(btn => btn.classList.remove('active'));
                    this.classList.add('active');
                });
            });

            // Mettre à jour le bouton actif basé sur la position de défilement
            window.addEventListener('scroll', function() {
                let currentSectionId = '';
                const sections = document.querySelectorAll('.section-block');
                const scrollY = window.pageYOffset || document.documentElement.scrollTop;

                sections.forEach(section => {
                    // Le haut de la section est détecté quand il passe SCROLL_OFFSET_SPY pixels du haut de la fenêtre
                    const sectionTop = section.offsetTop - SCROLL_OFFSET_SPY;
                    const sectionBottom = sectionTop + section.offsetHeight;

                    if (scrollY >= sectionTop && scrollY < sectionBottom) {
                        currentSectionId = section.getAttribute('id').replace('section-', '');
                    }
                });

                document.querySelectorAll('.section-button').forEach(button => {
                    button.classList.remove('active');
                    if (button.getAttribute('data-section') === currentSectionId) {
                        button.classList.add('active');
                    }
                });
            });

        function convertMarkdownToHTML(markdown) {
            if (typeof marked !== 'undefined') {
                return marked.parse(markdown);
            } else {
                console.warn("marked.js n'est pas chargé. Utilisation d'un fallback basique pour Markdown.");
                return markdown
                    .replace(/^# (.*$)/gim, '<h3>$1</h3>')
                    .replace(/^## (.*$)/gim, '<h4>$1</h4>')
                    .replace(/^### (.*$)/gim, '<h5>$1</h5>')
                    .replace(/^\* (.*$)/gim, '<li>$1</li>')
                    .replace(/^- (.*$)/gim, '<li>$1</li>')
                    .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
                    .replace(/(\n(?:[\s\t]*?(?:-|\*|\d\.) .*)+)/g, function(match) {
                        const listType = match.match(/^\n[\s\t]*?(\*|-)/) ? 'ul' : 'ol';
                        const items = match.trim().split('\n').map(item => `<li>${item.replace(/^[\s\t]*?(?:-|\*|\d\.) /, '')}</li>`).join('');
                        return `<${listType}>${items}</${listType}>`;
                    })
                    .replace(/\n\n/g, '</p><p>') // Paragraphes
                    .replace(/^(?!<[houlப்tabletrtd])(.*?)(\n|$)/gim, (match, p1) => p1.trim() === '' ? '' : `<p>${p1.trim()}</p>`) // Encapsuler lignes seules dans <p>
                    .replace(/<p><\/p>/g, ''); // Nettoyer paragraphes vides
            }
        }

                lastScrollTop = scrollTop <= 0 ? 0 : scrollTop;
            });
    </script>
